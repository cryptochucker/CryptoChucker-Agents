//@version=6
indicator("CryptoChucker - Money Scanner v1", overlay=false, max_labels_count=500)

// ============================================================================
// CryptoChucker Money Scanner v1
//
// Screener table that shows Money Line state (BULL/BEAR), a strength proxy,
// and whether the last bar was a flip for up to 30 user-configurable symbols.
//
// The Money Line computation mirrors indicators/money_line_pine.txt:
//   tp   = (high + low + close) / 3
//   vwma = sum(tp * volume, length) / sum(volume, length)
//   ml   = ema(vwma, smooth)
//   slope = sma(change(ml), slopeLen)
//   bull  = slope >= 0
//
// Strength proxy (0-100, no pandas_ta required in Pine):
//   atr_val  = ta.atr(length)           -- per-symbol ATR
//   slope_norm = |slope| / atr_val       -- normalized directional momentum
//   strength   = 100 * tanh(slope_norm * 8)  -- squashed to 0-100
//
// The scanner is capped at exactly 30 input.symbol slots to stay within
// TradingView's per-indicator request.security budget.
// ============================================================================

// ---- Shared inputs ---------------------------------------------------------
length   = input.int(8,  "VWMA length",   minval=1, group="Money Line Parameters")
smooth   = input.int(14, "EMA smooth",    minval=1, group="Money Line Parameters")
slopeLen = input.int(3,  "Slope length",  minval=1, group="Money Line Parameters")

// ---- 30 symbol inputs ------------------------------------------------------

[bull01, str01, flip01] = request.security(sym01, timeframe.period,
    [ta.sma(ta.change(ta.ema(math.sum((high+low+close)/3.0*volume,length)/math.sum(volume,length),smooth)),slopeLen)>=0,
     math.min(100.0,100.0*math.tanh((ta.atr(14)!=0?math.abs(ta.sma(ta.change(ta.ema(math.sum((high+low+close)/3.0*volume,length)/math.sum(volume,length),smooth)),slopeLen))/ta.atr(14):0.0)*8.0)),
     (not na(ta.sma(ta.change(ta.ema(math.sum((high+low+close)/3.0*volume,length)/math.sum(volume,length),smooth)),slopeLen)[1]) and bar_index>smooth+slopeLen) and
     (ta.sma(ta.change(ta.ema(math.sum((high+low+close)/3.0*volume,length)/math.sum(volume,length),smooth)),slopeLen)>=0) !=
     (ta.sma(ta.change(ta.ema(math.sum((high+low+close)/3.0*volume,length)/math.sum(volume,length),smooth)),slopeLen)[1]>=0)],
    lookahead=barmerge.lookahead_off)
