// ---- Money Line slope for the requested symbol's context -------------------
mlSlope() =>
    tp   = (high + low + close) / 3.0
    vwma = math.sum(tp * volume, length) / math.sum(volume, length)
    ml   = ta.ema(vwma, smooth)
    ta.sma(ta.change(ml), slopeLen)

// ---- tanh via math.exp (Pine v6 has no math.tanh), overflow-clamped --------
_tanh(x) =>
    xc = math.max(-15.0, math.min(15.0, x))
    e2 = math.exp(2.0 * xc)
    (e2 - 1.0) / (e2 + 1.0)

_strength(sl, at) => at != 0 ? math.min(100.0, 100.0 * _tanh(math.abs(sl) / at * 8.0)) : 0.0
